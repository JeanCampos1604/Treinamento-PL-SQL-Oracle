SELECT FSNFSAIDA.NOTAFISCAL "Número da Nota",
       FSNFSAIDA.SERIE "Série da Nota",
       FSMODELONF.DESCRICAO "Modelo da Nota",
       FSNFSAIDA.DTEMISSAO "Data de Emissão",
       FSNFITEM.CODPRODUTO "Código do Item" ,
       FSPRODUTO.DESCRICAO "Descrição do Item",
       NVL(TBICMS.VALORIMPOSTO, 0) ICMS,
       NVL(TBIPI.VALORIMPOSTO, 0) IPI,
       NVL(TBPIS.VALORIMPOSTO, 0) PIS,
       NVL(TBCOFINS.VALORIMPOSTO, 0) COFINS,
       NVL(TBOUTROS.VALORIMPOSTO, 0) OUTROSIMPOSTOS,
       FSNFITEM.VALORTOTAL "Valor sem impostos",
       FSNFITEM.VALORTOTAL + NVL(TBICMS.VALORIMPOSTO, 0) + NVL(TBIPI.VALORIMPOSTO, 0) + NVL(TBPIS.VALORIMPOSTO, 0) + NVL(TBCOFINS.VALORIMPOSTO, 0) + NVL(TBOUTROS.VALORIMPOSTO, 0) "Valor com Impostos",
       GEEMPRESA.RAZAOSOCIAL "Nome da Empresa",
       GEESTADO.NOME "Estado do endereço comercial da Empresa"
  FROM FSNFSAIDA,
       FSMODELONF,
       FSNFITEM,
       FSPRODUTO,
       GEEMPRESA,
       GEENDEMPRESA,
       GEENDERECO,
       GEESTADO,
       (SELECT FSNFITEMIMP.INDICEITEM, 
               FSNFITEMIMP.SEQUENCIA, 
               FSNFITEMIMP.CODESTAB, 
               FSNFITEMIMP.VALORIMPOSTO 
          FROM FSNFITEMIMP, 
               FSIMPOSTOS 
         WHERE FSIMPOSTOS.CODIMPOSTO = FSNFITEMIMP.CODIMPOSTO 
           AND FSIMPOSTOS.CLASSIFICACAO = 0) TBICMS,
       (SELECT FSNFITEMIMP.INDICEITEM, 
               FSNFITEMIMP.SEQUENCIA, 
               FSNFITEMIMP.CODESTAB, 
               FSNFITEMIMP.VALORIMPOSTO 
          FROM FSNFITEMIMP, 
               FSIMPOSTOS 
         WHERE FSIMPOSTOS.CODIMPOSTO = FSNFITEMIMP.CODIMPOSTO 
           AND FSIMPOSTOS.CLASSIFICACAO = 3) TBIPI,
       (SELECT FSNFITEMIMP.INDICEITEM, 
               FSNFITEMIMP.SEQUENCIA, 
               FSNFITEMIMP.CODESTAB, 
               FSNFITEMIMP.VALORIMPOSTO 
          FROM FSNFITEMIMP, 
               FSIMPOSTOS 
         WHERE FSIMPOSTOS.CODIMPOSTO = FSNFITEMIMP.CODIMPOSTO 
           AND FSIMPOSTOS.CLASSIFICACAO = 4) TBPIS,
       (SELECT FSNFITEMIMP.INDICEITEM, 
               FSNFITEMIMP.SEQUENCIA, 
               FSNFITEMIMP.CODESTAB, 
               FSNFITEMIMP.VALORIMPOSTO 
          FROM FSNFITEMIMP, 
               FSIMPOSTOS 
         WHERE FSIMPOSTOS.CODIMPOSTO = FSNFITEMIMP.CODIMPOSTO 
           AND FSIMPOSTOS.CLASSIFICACAO = 6) TBCOFINS,
        (SELECT FSNFITEMIMP.INDICEITEM, 
               FSNFITEMIMP.SEQUENCIA, 
               FSNFITEMIMP.CODESTAB, 
               FSNFITEMIMP.VALORIMPOSTO 
          FROM FSNFITEMIMP, 
               FSIMPOSTOS 
         WHERE FSIMPOSTOS.CODIMPOSTO = FSNFITEMIMP.CODIMPOSTO 
           AND FSIMPOSTOS.CLASSIFICACAO NOT IN (1, 3, 4, 6)) TBOUTROS
 WHERE FSNFSAIDA.CODESTAB = FSNFITEM.CODESTAB
   AND FSNFSAIDA.INDICENFSAI = FSNFITEM.INDICENFSAI
   AND FSMODELONF.CODMODELO = FSNFSAIDA.CODMODELO 
   AND FSPRODUTO.CODPRODUTO = FSNFITEM.CODPRODUTO
   AND TBICMS.CODESTAB   (+)= FSNFITEM.CODESTAB
   AND TBICMS.INDICEITEM (+)= FSNFITEM.INDICEITEM
   AND TBICMS.SEQUENCIA  (+)= FSNFITEM.SEQUENCIA
   AND TBIPI.CODESTAB   (+)= FSNFITEM.CODESTAB
   AND TBIPI.INDICEITEM (+)= FSNFITEM.INDICEITEM
   AND TBIPI.SEQUENCIA  (+)= FSNFITEM.SEQUENCIA
   AND TBPIS.CODESTAB   (+)= FSNFITEM.CODESTAB
   AND TBPIS.INDICEITEM (+)= FSNFITEM.INDICEITEM
   AND TBPIS.SEQUENCIA  (+)= FSNFITEM.SEQUENCIA
   AND TBCOFINS.CODESTAB   (+)= FSNFITEM.CODESTAB
   AND TBCOFINS.INDICEITEM (+)= FSNFITEM.INDICEITEM
   AND TBCOFINS.SEQUENCIA  (+)= FSNFITEM.SEQUENCIA
   AND TBOUTROS.CODESTAB   (+)= FSNFITEM.CODESTAB
   AND TBOUTROS.INDICEITEM (+)= FSNFITEM.INDICEITEM
   AND TBOUTROS.SEQUENCIA  (+)= FSNFITEM.SEQUENCIA
   AND GEEMPRESA.CODEMPRESA = FSNFSAIDA.CODEMPRESA
   AND GEEMPRESA.CODEMPRESA = GEENDEMPRESA.CODEMPRESA
   AND GEENDERECO.CODENDERECO = GEENDEMPRESA.CODENDERECO
   AND GEENDERECO.TIPO = 0
   AND GEENDERECO.CODESTADO = GEESTADO.CODESTADO
   AND FSNFSAIDA.NOTAFISCAL = 107509
   AND FSNFITEM.CODPRODUTO = '700.00053'
